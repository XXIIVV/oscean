#include <stdio.h>

typedef signed char Sint8;
typedef unsigned char Uint8;
typedef unsigned short Uint16;

typedef struct {
	Uint16 pc;
	Uint8 ptr[2];
	Uint8 stk[2][0x100];
	Uint8 d, r, k;
} Uxn;

static Uxn u;
static Uint16 console_vector;
static Uint8 ram[0x10000];
static Uint8 dev[0x100];

static Uint16 rx, ry, rz;

/*
@|IO */

static Uint8
dei(Uint8 port)
{
	return dev[port];
}

static void
deo(Uint8 port, Uint8 val)
{
	dev[port] = val;
	switch(port) {
	case 0x11:
		console_vector = dev[0x10] << 8 | val;
		return;
	case 0x18:
		fputc(val, stdout);
		return;
	case 0x19:
		fputc(val, stderr);
		return;
	}
}

/* clang-format off */

#define REC(m1, m2) if(!u.k) u.ptr[u.r] -= u.d ? m2 : m1; 
#define FIX(m) if(!u.k) u.ptr[u.r] -= m;

/* clang-format on */

/*
@|Primitives */

static inline Uint8
get8(int o)
{
	Uint8 offset = u.ptr[u.r] - o;
	return u.stk[u.r][offset];
}

static inline Uint16
get16(int o)
{
	Uint8 offset = u.ptr[u.r] - o;
	return (u.stk[u.r][offset] << 8 | u.stk[u.r][offset + 1]);
}

static inline Uint16
getx(int o8, int o16)
{
	if(u.d)
		return get16(o16);
	else
		return get8(o8);
}

static void
push8(Uint8 v)
{
	u.stk[u.r][u.ptr[u.r]++] = v;
}

static void
push(Uint16 v)
{
	if(u.d)
		push8((v >> 8));
	push8(v);
}

/*
@|Immediates */

static void
op_jci(void)
{
	rx = ram[u.pc] << 8 | ram[u.pc + 1];
	u.pc += 2;
	if(u.stk[0][--u.ptr[0]])
		u.pc += rx;
}

static void
op_jmi(void)
{
	rx = ram[u.pc] << 8 | ram[u.pc + 1];
	u.pc += 2 + rx;
}

static void
op_jsi(void)
{
	rx = ram[u.pc] << 8 | ram[u.pc + 1];
	u.pc += 2;
	u.r = 1;
	push8((u.pc >> 8));
	push8(u.pc);
	u.pc += rx;
}

static void
op_lit(void)
{
	u.r = 0;
	push8(ram[u.pc++]);
}

static void
op_li2(void)
{
	u.r = 0;
	push8(ram[u.pc++]);
	push8(ram[u.pc++]);
}

static void
op_lir(void)
{
	u.r = 1;
	push8(ram[u.pc++]);
}

static void
op_l2r(void)
{
	u.r = 1;
	push8(ram[u.pc++]);
	push8(ram[u.pc++]);
}

/*
@|Opcodes */

static void
op_inc(void)
{
	rx = getx(1, 2);
	REC(1, 2)
	push(rx + 1);
}

static void
op_pop(void)
{
	REC(1, 2)
}

static void
op_nip(void)
{
	rx = getx(1, 2);
	REC(2, 4)
	push(rx);
}

static void
op_swp(void)
{
	rx = getx(1, 2);
	ry = getx(2, 4);
	REC(2, 4)
	push(rx);
	push(ry);
}

static void
op_rot(void)
{
	rx = getx(1, 2);
	ry = getx(2, 4);
	rz = getx(3, 6);
	REC(3, 6)
	push(ry);
	push(rx);
	push(rz);
}

static void
op_dup(void)
{
	rx = getx(1, 2);
	REC(1, 2)
	push(rx);
	push(rx);
}

static void
op_ovr(void)
{
	rx = getx(1, 2);
	ry = getx(2, 4);
	REC(2, 4)
	push(ry);
	push(rx);
	push(ry);
}

static void
op_equ(void)
{
	rx = getx(1, 2);
	ry = getx(2, 4);
	REC(2, 4)
	push8(rx == ry);
}

static void
op_neq(void)
{
	rx = getx(1, 2);
	ry = getx(2, 4);
	REC(2, 4)
	push8(ry != rx);
}

static void
op_gth(void)
{
	rx = getx(1, 2);
	ry = getx(2, 4);
	REC(2, 4)
	push8(ry > rx);
}

static void
op_lth(void)
{
	rx = getx(1, 2);
	ry = getx(2, 4);
	REC(2, 4)
	push8(ry < rx);
}

static void
op_jmp(void)
{
	rx = getx(1, 2);
	REC(1, 2)
	u.pc = u.d ? rx : u.pc + (Sint8)rx;
}

static void
op_jcn(void)
{
	rx = getx(1, 2);
	ry = get8(u.d ? 3 : 2);
	REC(2, 3)
	if(ry)
		u.pc = u.d ? rx : u.pc + (Sint8)rx;
}

static void
op_jsr(void)
{
	rx = getx(1, 2);
	REC(1, 2)
	u.r = !u.r;
	push8((u.pc >> 8));
	push8(u.pc);
	u.pc = u.d ? rx : u.pc + (Sint8)rx;
}

static void
op_sth(void)
{
	rx = getx(1, 2);
	REC(1, 2)
	u.r = !u.r;
	push(rx);
}

static void
op_ldz(void)
{
	rx = get8(1);
	FIX(1)
	push8(ram[rx]);
	if(u.d)
		push8(ram[(Uint8)(rx + 1)]);
}

static void
op_stz(void)
{
	rx = get8(1);
	ry = getx(2, 3);
	REC(2, 3)
	if(u.d)
		ram[rx] = ry >> 8, ram[(Uint8)(rx + 1)] = ry;
	else
		ram[rx] = ry;
}

static void
op_ldr(void)
{
	rx = get8(1);
	ry = u.pc + (Sint8)rx;
	FIX(1)
	push8(ram[ry]);
	if(u.d)
		push8(ram[(Uint16)(ry + 1)]);
}

static void
op_str(void)
{
	rx = get8(1);
	ry = getx(2, 3);
	rz = u.pc + (Sint8)rx;
	REC(2, 3)
	if(u.d)
		ram[rz] = ry >> 8, ram[(Uint16)(rz + 1)] = ry;
	else
		ram[rz] = ry;
}

static void
op_lda(void)
{
	rx = get16(2);
	FIX(2)
	push8(ram[rx]);
	if(u.d)
		push8(ram[(Uint16)(rx + 1)]);
}

static void
op_sta(void)
{
	rx = get16(2);
	ry = getx(3, 4);
	REC(3, 4)
	ram[rx] = u.d ? ry >> 8 : ry;
	if(u.d)
		ram[(Uint16)(rx + 1)] = ry;
}

static void
op_dei(void)
{
	rx = get8(1);
	FIX(1)
	push8(dei(rx));
	if(u.d)
		push8(dei(rx + 1));
}

static void
op_deo(void)
{
	rx = get8(1);
	Uint16 ry = getx(2, 3);
	REC(2, 3)
	if(u.d)
		deo(rx, ry >> 8);
	deo(u.d ? rx + 1 : rx, ry);
}

static void
op_add(void)
{
	rx = getx(1, 2);
	ry = getx(2, 4);
	REC(2, 4)
	push(ry + rx);
}

static void
op_sub(void)
{
	rx = getx(1, 2);
	ry = getx(2, 4);
	REC(2, 4)
	push(ry - rx);
}

static void
op_mul(void)
{
	rx = getx(1, 2);
	ry = getx(2, 4);
	REC(2, 4)
	push(ry * rx);
}

static void
op_div(void)
{
	rx = getx(1, 2);
	ry = getx(2, 4);
	REC(2, 4)
	push(rx ? ry / rx : 0);
}

static void
op_and(void)
{
	rx = getx(1, 2);
	ry = getx(2, 4);
	REC(2, 4)
	push(ry & rx);
}

static void
op_ora(void)
{
	rx = getx(1, 2);
	ry = getx(2, 4);
	REC(2, 4)
	push(ry | rx);
}

static void
op_eor(void)
{
	rx = getx(1, 2);
	ry = getx(2, 4);
	REC(2, 4)
	push(ry ^ rx);
}

static void
op_sft(void)
{
	rx = get8(1);
	ry = getx(2, 3);
	REC(2, 3)
	push((ry >> (rx & 0x0f)) << (rx >> 4));
}

typedef void (*Opc)(void);

/* clang-format off */

static const Opc imm[8] = {
	NULL,   op_jci, op_jmi, op_jsi,
	op_lit, op_li2, op_lir, op_l2r};

static const Opc ops[32] = {
	NULL,   op_inc, op_pop, op_nip,
	op_swp, op_rot, op_dup, op_ovr,
	op_equ, op_neq, op_gth, op_lth,
	op_jmp, op_jcn, op_jsr, op_sth,
	op_ldz, op_stz, op_ldr, op_str,
	op_lda, op_sta, op_dei, op_deo,
	op_add, op_sub, op_mul, op_div,
	op_and, op_ora, op_eor, op_sft};

/* clang-format on */

/*
@|Core */

static unsigned int
uxn_eval(Uint16 start_pc)
{
	u.pc = start_pc;
	for(;;) {
		Uint8 instr = ram[u.pc++];
		Uint8 op = instr & 0x1f;
		if(!op) {
			Uint8 idx = instr >> 5;
			if(!idx)
				return 1;
			imm[idx]();
		} else {
			u.d = (instr >> 5) & 1;
			u.r = (instr >> 6) & 1;
			u.k = instr >> 7;
			ops[op]();
		}
	}
	return 0;
}

static void
console_input(int c, unsigned int type)
{
	dev[0x12] = (Uint8)c;
	dev[0x17] = (Uint8)type;
	if(console_vector)
		uxn_eval(console_vector);
}

int
main(int argc, char **argv)
{
	FILE *f;
	if(argc < 2)
		return 1;
	else if(!(f = fopen(argv[1], "rb")))
		return 1;
	fread(&ram[0x100], 0xff00, 1, f), fclose(f);
	dev[0x17] = argc > 2;
	if(uxn_eval(0x100) && console_vector) {
		int i = 2;
		for(; i < argc; i++) {
			char c, *p = argv[i];
			while(!dev[0x0f] && (c = *p++))
				console_input(c, 2);
			console_input('\n', 3 + (i == argc - 1));
		}
		while(!dev[0x0f]) {
			char c = fgetc(stdin);
			if(feof(stdin)) break;
			console_input(c, 1);
		}
		console_input('\n', 4);
	}
	return dev[0x0f] & 0x7f;
}
