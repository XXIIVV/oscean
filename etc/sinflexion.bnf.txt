<verb-stem>     ::= INVARIANT-STEM
<contraction>   ::= VOWEL "'" SAME-VOWEL...

<negation>      ::= "non" | "ne"
<past>          ::= "jam" | "tum" | "e"
<future>        ::= "vol" | "deb" | "i"
<tense>         ::= <past> | <future>
<modal>         ::= "pot"

<pronoun>       ::= "me" | "te" | "il" | "nos" | "vos" | "ils" | "id" | "se"
<possessive>    ::= "de" <pronoun>

<demonstrative> ::= "hoc" | "il" | "ils"

<indefinite>    ::= "tal" | "qual" | "omn" | "ul" | "nul" | "nemin"
                   | "al" | "sol" | "tot" | "neutr" | "alter" | "utroq"
                   | "aliq" | "alioq" | "aliquid" | "aliquand" | "aliquant"
<interrog>      ::= "quomod" | "ubi" | "quand" | "quant" | "qui"

<cardinal>      ::= "un" | "du" | "tres" | "quatuor" | "quinq" | "sex"
                   | "septem" | "oct" | "novem" | "decem"
                   | "vigint" | "trigint" | "quadragint" | "quinquagint"
                   | "sexagint" | "septuagint" | "octogint" | "nonagint"
                   | "centum" | "mil" | "milion"
<ordinal>       ::= "prim" | "secund" | "tert" | "quart" | "quint"
                   | "sext" | "septim" | "octav" | "non" | "decim"
<fraction>      ::= "dimid" | "un" <ordinal>
<compound-num>  ::= <cardinal> "-" <cardinal>
<numeral>       ::= <cardinal> | <ordinal> | <fraction> | <compound-num>

<coord-conj>    ::= "et" | "aut" | "vel" | "sed" | "nec" | "neq"
<conjunction>   ::= "si" | "que" | "quod" | <coord-conj>





<verb-phrase>   ::= <negation>? <tense>? <modal>? <verb-stem>
                     ; TODO: <adverb> slot — verb-phrase only, or noun-phrase too?

<genitive>      ::= "de" <noun-phrase>
<adjective>     ::= <genitive> | ADJ-STEM
<adverb>        ::= ADV-STEM | "in" "mod" <adjective>

<noun>   ::= <pronoun> | <indefinite>
                   | <numeral>? <demonstrative>? <noun> <genitive>? <adjective>?
                     ; TODO: numeral standalone ("Quant es? Centum") — own alternative?

<comparison>    ::= ADJ
                   | "magis" ADJ "quam" <noun-phrase>
                   | "maxim" ADJ
                   | "tam" ADJ "quam" <noun-phrase>

<element>       ::= <noun-phrase> | <verb-phrase> | <adjective>
<correlative>   ::= "aut" <element> "aut" <element>
                   | "et" <element> "et" <element>
                   | "neq" <element> "nec" <element>
                   | "minus" <element> "quam" <element>
                   | "plus" <element> "quam" <element>
                   | "tant" <element> "quant" <element>
                   | "qual" <element> "it" <element>
                   | "vel" <element> "vel" <element>
                     ; TODO: where does <correlative> plug into <sentence>?
                     
<subordinate>   ::= <conjunction> <sentence>

<declarative>   ::= ; Leon dorm in umbr.
                      <noun> <verb> <prep> <noun>
                    ; Monday we will read
                    | <time> <noun> <verb> <noun>

<interrogative> ::= ; Quomod es te? 
                      <interrog> <verb> <noun>
                    ; E dorm il in umbr?
                    | <verb-aux> <verb> <noun> <prep> <noun>
                  
<imperative>    ::= ; Dic ce de nov.
                      <verb> <noun>?
                    ; D'ad me librs.
                    | <verb> <prep> <noun> <noun>

<sentence>      ::= <declarative> | <interrogative> | <imperative> | <passive>
