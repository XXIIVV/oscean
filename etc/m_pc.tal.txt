@input/on-controller ( -> )
	.Controller/button DEI2 DUP2 manifest/scan INC2k ORA ?{
		POP2
		( key ) /handle-input
		( button ) POP BRK }
	( * ) JSR2 POP2 BRK

@input/on-console ( -> )
	.Console/read DEI
	( | mute arg spacers )
	[ LIT2 03 -Console/type ] DEI NEQ ?{ POP #20 }
	/handle-input BRK

@input/handle-input ( c -- )
	DUP ?{ POP JMP2r }
	DUP #08 EQU ?/on-bs
	DUP #0a EQU ?/on-enter
	DUP #7f EQU ?/on-del
	DUP #20 GTH ?{ POP #20 }
	/<push> !/<redraw>

	&on-bs ( c -- )
	POP /<pop> !/<redraw>

	&on-enter ( c -- )
	POP !/<eval>

	&on-del ( c -- )
	POP !/<blank>

@input/<reset> ( -- )
	#0000 .&buf STZ2
	[ LIT2 -&buf _&ptr ] STR
	JMP2r

@input/<push-str> ( str* -- )
	&>w
		LDAk DUP ?{ POP POP2 !/<redraw> }
		/<push>
		INC2 !/>w

@input/<push> ( c -- )
	[ LIT2 00 &ptr -&buf ] INCk ,&ptr STR
	STZ2
	JMP2r

@input/<pop> ( -- )
	,&ptr LDR
	( cap ) DUP .&buf NEQ ?{ POP JMP2r }
	#01 SUB #00 OVR STZ
	,&ptr STR
	JMP2r

@input/<eval> ( -- )
	;&buf LDAk ?{ POP2 /<reset> !bios/<draw-welcome> }
	bios/<parse>
	/<reset> !/<redraw>

@input/<eval-partial> ( -- )
	;&buf LDAk ?{ POP2 /<reset> !bios/<draw-welcome> }
	bios/<parse-partial>
	/<reset> !/<redraw>

@input/<blank> ( -- )
	sel/<reset>
	/<reset>
	bios/<draw>
	( >> )

@path/is-folder ( -- t )
	,&ptr LDR2 #0001 SUB2 LDA LIT "/ EQU JMP2r

@path/<move> ( str* -- )
	DUP2 ;dict/out-path str/cmp ?/<pop>
	( | Check if file is a folder )
	LITr _&ptr LDR2r /<push-str>
	;&buf file/is-folder ?{
		#00 STH2kr STA
		LITr _&ptr STR2r
		JMP2r }
	POP2r
	( | Add missing slash )
	/is-folder ?{ LIT "/ /<push> }
	JMP2r

@path/<pop> ( str* -- )
	POP2
	( | find last slash )
	;&buf ,&ptr LDR2 #0001 SUB2
	( clamp ) DUP2 ;&buf INC2 NEQ2 ?{ POP2 POP2 JMP2r }
	&>lp
		#0001 SUB2 LDAk LIT "/ EQU ?{ LTH2k ?/>lp }
	INC2 #0000 SWP2 STA2
	POP2 JMP2r

@sel/<prev> ( -- )
	,&id LDR

	#01 SUB !/<set>

@cmds/find ( str* -- <cmd>* )
	,&fn STR2
	;&lutend ;&lut
	&>l
		LDA2k LDA2 [ LIT2 &fn $2 ] str/cmp ?{
			INC2 INC2 GTH2k ?/>l
		POP2 #ffff }
	NIP2 JMP2r

@cmds/lut [ =cmd/api ]
	&lut-clip [
	=dir/api =mov/api =now/api =run/api
	=icn/api =len/api =put/api =get/api
	=cpy/api =era/api =ren/api =pop/api
	=dup/api =ovr/api =swp/api =dec/api
	=hex/api =cat/api =cmp/api =and/api
	=ora/api =out/api =bye/api &lutend ]

@cmd/<eval> ( -- )
	;cmds/lutend ;cmds/lut-clip
	&>l
		LDA2k #0004 ADD2 bios/<push-raw>
		INC2 INC2 GTH2k ?/>l
	POP2 POP2 JMP2r
	&api =dict/cmd =&<eval> "<cmd> 20 "( 20 09 20 ") 00

@dir/<eval> ( -- )
	;path/buf .File/name DEO2
	;&buflen .File/length DEO2
	;&buf .File/read DEO2
	;&buf DUP2 .File/success DEI2 ADD2 SWP2

		LDAk ?{ POP2 POP2 JMP2r }
		#0005 ADD2 dict/alloc-line bios/<push-raw>
		INC2 !/>l
	&api =dict/dir =&<eval> "<dir> 20 "( 20 09 20 "[f] 20 ") 00

@icn/<draw> ( name* -- )
	DUP2 .File/name DEO2

	( | size )

	( h ) #00 SWP SUB ,&height STR

	.Screen/y DEO2
	( | draw )
	[ LIT2 01 -Screen/auto ] DEO
	[ LIT &height $1 ]
	&>v
		#0000 .Screen/x DEO2
		[ LIT &width $1 ]
		&>h
			;&buf .File/read DEO2
			[ LIT2 00 -File/success-lb ] DEI EQU ?{ DEOkr
				INC DUP ?/>h }
		POP
		( lb ) .Screen/y DEI2 #0008 ADD2 .Screen/y DEO2
		INC DUP ?/>v
	POP POP2r JMP2r

@icn/read-size ( str* -- w h )

(
@|Dict )

@dict/alloc-cat ( a* b* -- addr* )
	[ LIT2r =&ptr ] LDA2r
	&>wc1
		LDAk DUP ?{
			POP POP2
			&>wc2
				LDAk DUP ?{
					POP POP2 STH2r #00 !/<push> }
				/<push>
				INC2 !/>wc2 }
		/<push>
		INC2 !/>wc1

@dict/alloc ( str* -- str* addr* )
	LDAk LIT "" EQU ?/alloc-string
	[ LITr _&ptr ] LDR2r
	&>w
		LDAk DUP #20 GTH ?{ POP STH2r #00 !/<push> }
		/<push>
		INC2 !/>w

@dict/alloc-string ( str* -- str* addr* )
	INC2 [ LITr _&ptr ] LDR2r
	&>was
		LDAk DUP ?{ &end POP INC2 STH2r #00 !/<push> }
		DUP LIT "" EQU ?/end
		/<push>
		INC2 !/>was

@dict/alloc-line ( str* -- str* addr* )
	[ LITr _&ptr ] LDR2r
	&>wal
		LDAk DUP #1f GTH ?{ POP STH2r #00 !/<push> }
		/<push>
		INC2 !/>wal

		INC2 !/>waf
